对大矩阵循环，计算每个待比较项目的的p值。
{% highlight r %}
list = seq(1, 126, by = 2)

for(i in list){
  # print(re_inner[,i:(i+1)])
  print(fisher.test(re_data[,i:(i+1)])$p.value)
}
{% endhighlight %}
